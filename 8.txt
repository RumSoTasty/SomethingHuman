#include <math.h>
#include <iostream>
#include <stdlib.h>
using namespace std;
int main()
{

        cout<<"Task 8.1: "<<endl;
        float x,v1,v2,a1,a2,t,s1,s2;
        cout<<"Rasstoianie mezdy mashinami: ";
        cin>>x;
        cout<<"Vvedite skorost pervogo avtomobila: ";
        cin>>v1;
        cout<<"Vvedite skorost vtorogo avtomobila: ";
        cin>>v2;
        cout<<"Vvedite uskorenie pervogo avtomobila: ";
        cin>>a1;
        cout<<"Vvedite uskorenie vtorogo avtomobila: ";
        cin>>a2;
        t=(-(v1+v2)+sqrt((v1+v2)*(v1+v2)+(a1+a2)*2*x))/(a1+a2);
        cout<<"Vrema vstrechi: "<<t<<endl;

        cout<<"Task 8.2: "<<endl;
        s1=((a1*(t*t))/2)+v1*t;
        s2=((a2*(t*t))/2)+v2*t;

        cout<<"Rasstoinaie proidenoe vtorim avtomobilem: "<<s2<<endl;

        cout<<"Vvedite cenu za 1 litr topliva: ";
        cin>>x;

        a1=(s1*v1)/x;
        a2=(s2*v2)/x;

        cout<<"Obyom zatrachevaemogo topliva pervogo avtomobila: "<<a1<<", Czena za toplivo: "<<a1*x<<endl;
        cout<<"Obyom zatrachevaemogo topliva vtorogo avtomobila: "<<a2<<", Czena za toplivo: "<<a2*x<<endl;

}
